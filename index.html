Git


>git init :ijade yek repository

>git status :vaziyat kononi repository ra neshan midahad

>git add -A :hameye taghirate anjam shode roye file ha ra ezafeh mikone be git

>git restore --staged :az git kharej mikone mokhalef ADD hastesh

>git add fileName :faghat file morede nazar ra ezafe mikone be git

>git commit -m "Comment" : taghirati ke ro ye file ha anjam shode ra ba ye comment moshakhas mikone

>git diff HEAD :taghirate ghab va bade file ra namayeah midahad

>git diff --staged : taghirat daron stage ra neshan midahad masalan xhe file hayi taghir kardan

>git reset fileName :az daron stage kharej mikonad file morede nazar ra vali taghirat ra negah midarad

>git checkout fileName :Taghirat ra az beyn mibarad

>git log : commit hayi ke anjam dadim ra namayesh midahad

>git branch :neshan midahad che branch haye vojod darad

>git branch BranchName : ijad mikonad yek branch jadid

>git check out branchName : miravad dar branch nam borde

>git merge BranchName : in dastor bayad daron branch master neveshte beshe va baes mishe taghirate injad shode be master ezefeh beshe

>git rm FileName :pak mikonad file ra ham az git ham ay directory

>git branch -d branchName : pak mikonad branche morede nazar ra

>git blame "filename" -l : file moredenazar ra baresi mikonad va neshan midahad ke che kasi che taghiri dar file ijad karde

>rm -rf .git :pak mikone repository

>ls -la : tamami directory ha va zir directory hara neshan midahad

>vi file name : file morede nazar ra baz mikonad va az tarigh :x zakhire va kharej mishavad az file

cat filename :file morede nazar ra faghat mikhanad
